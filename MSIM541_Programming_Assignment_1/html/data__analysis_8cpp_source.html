52 void computeNormalFunc(float mu, float sigma)
53 {
54  // This function computes the normal distribution and outputs to arrays
55  // Determine the step size and compute the arrays curveX and curveY
56  float stepSizeN = (7.0) / numCurvePoints; // 7.0 yields curve that extends from -3.5 to 3.5, good match for normal, mu=0, sigma=1
57  for (int i = 0; i < numCurvePoints; i++)
58  {
59  curveXn[i] = -3.5 + stepSizeN * i;

76 void testForDistro(float mu, float sigma, float lambda)
77 {
78  /*this function automatically determines which theoretical distribution is the best
79  fit for the input data. It uses a Jarque-Bera test for normality. Since there are only
80  normal and exponential distributions in this assignment, any curve failing this test
81  is assumed to be exponential fit.*/
82  // compute descriptive statistics for data file
83  float datafileSum = 0.0;
84  for (int i = 0; i < numDataPoints; i++)
85  {
86  datafileSum = datafileSum + data_ptr[i];
87  }
88  float fileMean = datafileSum / (float)numDataPoints;
89  // initialize values to be computed later
90  float fileVar = 0.0;
91  float fileStdDev = 0.0;
92  float fileSkewness = 0.0;
93  float fileKurtosis = 0.0;
94 
95  for (int j = 0; j < numDataPoints; j++)
96  {
97  fileVar += pow((data_ptr[j] - fileMean), 2);
98  fileSkewness += pow((data_ptr[j] - fileMean), 3);
99  fileKurtosis += pow((data_ptr[j] - fileMean), 4);
100  }
101  fileVar = fileVar / numDataPoints;
102  fileSkewness = (fileSkewness / numDataPoints) / pow(fileVar, 1.5);
103  fileKurtosis = (fileKurtosis / numDataPoints) / pow(fileVar, 2);
104 
105  // apply tests
106  // Jarque-Bera test
107  float JB = (numDataPoints / 6) * (pow(fileSkewness, 2) + 0.25 * pow((fileKurtosis - 3.0), 2));
108  if (JB < 3.0) // less than 3.0 fits standard normal distribution
109  {
110  isNormal = true;
111  }
112  else
113  isNormal = false;
114 }

268 void computeProbability(int numIntervals)
269 {
270  // Delete previously allocated memory
271  if (endPoints != NULL)
272  {
273  delete endPoints;
274  }
275 
276  if (prob != NULL)
277  {
278  delete prob;
279  }
280 
281  // establish arrays for endpoints and probabilities
282  endPoints = new float[numIntervals];
283  prob = new float[numIntervals];
284 
285  // Determine the end points for each interval (update the array endPoints)
286  float range = maximum - minimum;
287  for (int i = 0; i < numIntervals; i++)
288  {
289  endPoints[i] = minimum + ((range / numIntervals) * i);
290  }
291 
292  // Compute the probability for each interval (update the array prob)
293  for (int i = 0; i < numIntervals; i++)
294  {
295  int counter = 0;
296  for (int j = 0; j < numDataPoints; j++)
297  {
298  if (data_ptr[j] <= endPoints[i] && data_ptr[j] > endPoints[i-1])
299  {
300  counter += 1;
301  }
302  }
303  float pdf = (float)counter / numDataPoints;
304  prob[i] = pdf;
305  }
